import <contextTestsImport>

final class <nodeName>ContextTests: XCTestCase {

    private var analyticsMock: <nodeName>AnalyticsMock!
    private var context: <nodeName>ContextImp!

    override func setUp() {
        super.setUp()
        analyticsMock = <nodeName>AnalyticsMock()
        context = <nodeName>ContextImp(
            workers: [],
            analytics: analyticsMock
        )
    }

    override func tearDown() {
        analyticsMock = nil
        context = nil
        super.tearDown()
    }

    func testNotNil() {
        expect { self.context } != nil
    }

    func testWorkersIsEmpty() {
        expect { self.context.workers }.to(beEmpty())
    }

    func testActivation() {
        expect { self.context.isActive } == false
        context.activate()
        expect { self.context.isActive } == true
        context.deactivate()
        expect { self.context.isActive } == false
    }
}
