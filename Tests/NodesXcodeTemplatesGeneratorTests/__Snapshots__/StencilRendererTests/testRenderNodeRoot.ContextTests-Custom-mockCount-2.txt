import <testImport1>
import <testImport2>

final class RootContextTests: XCTestCase {

    private var analyticsMock: RootAnalyticsMock!
    private var context: RootContextImp!

    override func setUp() {
        super.setUp()
        analyticsMock = RootAnalyticsMock()
        context = RootContextImp(
            workers: [],
            analytics: analyticsMock
        )
    }

    override func tearDown() {
        analyticsMock = nil
        context = nil
        super.tearDown()
    }

    func testNotNil() {
        expect { self.context } != nil
    }

    func testWorkersIsEmpty() {
        expect { self.context.workers }.to(beEmpty())
    }

    func testActivation() {
        expect { self.context.isActive } == false
        context.activate()
        expect { self.context.isActive } == true
        context.deactivate()
        expect { self.context.isActive } == false
    }
}
