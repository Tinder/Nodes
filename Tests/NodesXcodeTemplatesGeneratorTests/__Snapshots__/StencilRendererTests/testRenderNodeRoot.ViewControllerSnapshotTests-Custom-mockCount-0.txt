final class RootViewControllerTests: ServiceTestCase {

    override func setUp() {
        super.setUp()
        recordMode = false
    }

    override func tearDown() {
        super.tearDown()
    }

    func testInitialState() {
        assertSnapshot(matching: RootViewController(viewState: publish(RootViewState())),
                       as: .image)
    }

    func testPublisher() {
        let publisher = PassthroughSubject<RootViewState, Never>().eraseToAnyPublisher()

        assertSnapshot(matching: RootViewController(viewState: publisher),
                       as: .image)
    }
}
